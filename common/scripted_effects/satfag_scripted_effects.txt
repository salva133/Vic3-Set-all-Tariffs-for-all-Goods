set_tariffs_to_no_tariffs = {
	root = {
		set_import_tariff_level = {
			goods = prev
			level = no_tariffs_or_subventions
		}
		set_export_tariff_level = {
			goods = prev
			level = no_tariffs_or_subventions
		}
	}
	custom_tooltip = set_tariffs_to_no_tariffs_tt
}
